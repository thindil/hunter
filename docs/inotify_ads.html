 Types of inotify events and values for them

   type Inotify_Events is
     (ACCESSED, MODIFIED, METADATA, CLOSED_WRITE, CLOSED_NO_WRITE, OPENED,
      MOVED_FROM, MOVED_TO, CREATED, DELETED, DELETED_SELF, MOVED_SELF,
